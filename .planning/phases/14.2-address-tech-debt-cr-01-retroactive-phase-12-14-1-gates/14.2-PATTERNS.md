**Canonical comparator signature** (`presentation/sort.ts:35-52`):
```typescript
interface Sortable {
  readonly name: string;
  readonly scope: "user" | "project";
}

export function compareByNameThenScope(a: Sortable, b: Sortable): number {
  const byName = a.name.localeCompare(b.name, undefined, { sensitivity: "base" });
  if (byName !== 0) {
    return byName;
  }

  // Tie-breaker: project before user per MSG-GR-3.
  if (a.scope === b.scope) {
    return 0;
  }

  return a.scope === "project" ? -1 : 1;
}
```

**`isUserFirstScopeOrder` helper sketch** (no analog -- new code; rationale: detect the body pattern `return scope === "user" ? <low> : <high>` and `return scope === "user" ? 0 : 1` specifically):
```javascript
function isUserFirstScopeOrder(body) {
  if (body.type !== "BlockStatement" || body.body.length !== 1) return false;
  const ret = body.body[0];
  if (ret?.type !== "ReturnStatement" || ret.argument?.type !== "ConditionalExpression") {
    return false;
  }
  const test = ret.argument.test;
  // test: scope === "user" (BinaryExpression, ===, identifier scope vs literal "user")
  if (
    test.type !== "BinaryExpression" ||
    test.operator !== "===" ||
    test.right?.type !== "Literal" ||
    test.right.value !== "user"
  ) {
    return false;
  }
  // consequent < alternate (user-first signal)
  const cons = ret.argument.consequent;
  const alt = ret.argument.alternate;
  if (cons.type !== "Literal" || alt.type !== "Literal") return false;
  return typeof cons.value === "number" && typeof alt.value === "number" && cons.value < alt.value;
}
```
