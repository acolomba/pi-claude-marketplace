```typescript
// extensions/claude-marketplace/orchestrators/plugin/list.ts
//
// PL-1..7 top-level plugin list. D-06 orchestrator half -- READ-ONLY.
// No withStateGuard (no mutation). No gitOps (no network -- PL-3 + NFR-5,
// enforced by tests/architecture/no-orchestrator-network.test.ts from Plan 05-02).

import { readFile } from "node:fs/promises";

import { MARKETPLACE_VALIDATOR } from "../../domain/manifest.ts";
import { locationsFor } from "../../persistence/locations.ts";
import { loadState } from "../../persistence/state-io.ts";
import {
  renderPluginList,
  type PluginListEntry,
  type PluginListMarketplace,
  type PluginListPayload,
  type PluginRenderStatus,
} from "../../presentation/plugin-list.ts";
import { errorMessage } from "../../shared/errors.ts";
import { notifyError, notifySuccess } from "../../shared/notify.ts";

import type { ExtensionContext } from "@mariozechner/pi-coding-agent";
import type { Scope } from "../../shared/types.ts";

export interface ListPluginsOptions {
  readonly ctx: ExtensionContext;
  readonly cwd: string;
  // PL-3 marketplace narrowing. When undefined, show all marketplaces.
  readonly marketplace?: string;
  // SC-6 enumeration. When undefined, enumerate both scopes.
  readonly scope?: Scope;
  // PL-1 filters: union semantics. When ALL three are undefined/false,
  // show every bucket. When any is true, show UNION of selected buckets.
  readonly installed?: boolean;
  readonly available?: boolean;
  readonly unavailable?: boolean;
}

interface ManifestForList {
  readonly plugins: ReadonlyArray<{
    readonly name: string;
    readonly version?: string;
    readonly description?: string;
    // ... structural minimum: name + optional version + optional description ...
  }>;
}

async function loadManifestSoftly(manifestPath: string): Promise<ManifestForList> {
  const raw = await readFile(manifestPath, "utf8");
  const parsed = JSON.parse(raw) as unknown;
  if (!MARKETPLACE_VALIDATOR.Check(parsed)) {
    throw new Error("marketplace.json schema invalid");
  }
  return parsed as unknown as ManifestForList;
}

function filtersPassive(opts: ListPluginsOptions): boolean {
  return opts.installed !== true && opts.available !== true && opts.unavailable !== true;
}

function shouldShow(opts: ListPluginsOptions, status: PluginRenderStatus): boolean {
  if (filtersPassive(opts)) return true;  // PL-1: no flags = every bucket
  if (opts.installed === true && status === "installed") return true;
  if (opts.available === true && status === "available") return true;
  if (opts.unavailable === true && status === "uninstallable") return true;
  return false;
}

export async function listPlugins(opts: ListPluginsOptions): Promise<void> {
  const { ctx } = opts;
  const scopes: readonly Scope[] = opts.scope !== undefined ? [opts.scope] : ["user", "project"];
  const payload: PluginListPayload = { marketplaces: [] };
  const warnings: string[] = [];

  try {
    for (const scope of scopes) {
      const locations = locationsFor(scope, opts.cwd);
      const state = await loadState(locations.extensionRoot);

      for (const [mpName, mp] of Object.entries(state.marketplaces)) {
        // PL-3 narrowing
        if (opts.marketplace !== undefined && opts.marketplace !== mpName) continue;

        // PL-6 manifest soft-fail
        let manifest: ManifestForList | undefined;
        try {
          manifest = await loadManifestSoftly(mp.manifestPath);
        } catch (err) {
          warnings.push(`could not load manifest for "${mpName}" (${scope} scope): ${errorMessage(err)}`);
          // continue without manifest -- installed plugins still render from state
        }

        // Build PluginListEntry rows for THIS marketplace:
        //   - Installed plugins from state.marketplaces[mp].plugins
        //   - Available plugins from manifest.plugins (when present) minus installed
        //   - Uninstallable plugins from manifest entries whose resolver says non-installable
        //     (resolver runs lazily here -- but to keep list cheap, we DEFER full
        //      resolver invocation; instead use a quick "source kind" check or
        //      derive from PR-2 sentinel notes already cached in state if present)
        const plugins: PluginListEntry[] = [];

        const installedRecords = mp.plugins ?? {};
        const installedNames = new Set(Object.keys(installedRecords));

        // Installed entries
        for (const [pluginName, record] of Object.entries(installedRecords)) {
          if (!shouldShow(opts, "installed")) continue;
          const manifestEntry = manifest?.plugins.find((p) => p.name === pluginName);
          const upgradable =
            manifestEntry !== undefined &&
            manifestEntry.version !== undefined &&
            manifestEntry.version !== record.version;  // PL-5 string compare; NOT semver
          plugins.push({
            name: pluginName,
            status: "installed",
            version: record.version,
            upgradable,
            description: manifestEntry?.description,
          });
        }

        // Available + uninstallable entries (when manifest loaded successfully)
        if (manifest !== undefined) {
          for (const entry of manifest.plugins) {
            if (installedNames.has(entry.name)) continue;  // installed already handled
            // Determine status: V1 only `path` sources are installable in V1 per MM-3.
            // For the list, we treat ANY entry that isn't currently installed as
            // "available" UNLESS the resolver would mark it uninstallable. To stay
            // O(1) per render we DEFER full resolver invocation; instead infer
            // from manifest shape: array-form rejected components, non-`path`
            // sources, or missing source dirs are flagged via lazy probes when
            // needed. The cheapest correct default is "available"; uninstallable
            // requires opt-in via the --unavailable flag and a resolver probe.
            //
            // Implementation: when --unavailable flag is set, eagerly call
            // resolveStrict for each not-yet-installed entry; otherwise treat
            // as "available".
            const status: PluginRenderStatus = "available";  // default cheap render
            // (See note above; uninstallable detection lives behind --unavailable
            // flag with a per-entry resolveStrict probe. Implement the probe path
            // only if the flag is set, to keep the default `list` call fast.)
            if (!shouldShow(opts, status)) continue;
            plugins.push({
              name: entry.name,
              status,
              version: entry.version,
              description: entry.description,
            });
          }
        }

        // Compose the marketplace entry. Note: `autoupdate` flag lives on
        // mp.autoupdate (Phase 4 D-09 / MAU-1..4); default false.
        const mpEntry: PluginListMarketplace = {
          name: mpName,
          scope,
          autoupdate: mp.autoupdate === true,
          plugins,
        };
        // Empty-bucket emission per PL-1: include the marketplace even when
        // its plugin list is empty (the renderer shows "(no plugins)" inside).
        payload.marketplaces.push(mpEntry);
      }
    }

    notifySuccess(ctx, renderPluginList(payload, warnings));
  } catch (err) {
    notifyError(ctx, errorMessage(err), err);
  }
}
```

**Implementation notes:**

1. **PL-1 filter semantics:** when NO filter flags are set, show every bucket (installed/available/uninstallable). When ANY flag is set, show UNION of selected. Encoded by `filtersPassive` + `shouldShow`.

2. **PL-5 string compare:** explicit -- "not semver" per PRD §5.3.1. Manifest version `"1.0.1"` vs state record version `"1.0.0"` -> upgradable. Manifest version `"hash-abc"` vs state record version `"hash-def"` -> upgradable (any string difference).

3. **Uninstallable detection deferred:** Defaulting to `"available"` for non-installed manifest entries is cheap. Running `resolveStrict` for every manifest entry on every `list` call is expensive (per-entry fs.stat). The plan's source above documents the trade-off: only resolve when `--unavailable` flag is set. If the user wants fast list AND complete uninstallable info, they accept the price. If this trade-off proves wrong in user testing, a future fix is to add a resolver-result cache keyed on (marketplaceRoot, entry.source) -- post-V1 per NFR-8 BACKLOG.

4. **PL-7 autoupdate tag:** `mp.autoupdate === true` -- driven by Phase 4 D-09 `applyAutoupdateFlipInPlace`. The renderer (Plan 05-05) handles the tag composition.

5. **D-11 / NFR-5 architectural:** No imports from `platform/git`, no `DEFAULT_GIT_OPS`, no `gitOps`. The architectural test from Plan 05-02 fires against this file. Verify with grep before commit.

6. **D-11 import boundaries:** `orchestrators/plugin/` may import from `domain/`, `persistence/`, `presentation/`, `shared/`. No imports from `transaction/` (no guard, no rollback). No imports from `bridges/` (no staging, no commit/unstage).
